## Minimum Operations to Make Median of Array Equal to K

<details>
<summary>Python</summary>

```python
class Solution:
    def minOperationsToMakeMedianK(self, nums: List[int], k: int) -> int:
        n = len(nums)
        ans = 0
        nums.sort()
        for i in range(n//2+1):
            ans+=max(0,nums[i]-k)
        for i in range(n//2,len(nums)):
            ans+=max(0,k-nums[i])
        return ans
```

```cpp
        for (int i = 0; i < n / 2 + 1; ++i) {
```

## Minimum Cost Walk in Weighted Graph

<details>
<summary>Python</summary>

```python
class Solution:
    def minimumCost(self, n: int, edges: List[List[int]], query: List[List[int]]) -> List[int]:
        gr = [[] for _ in range(n)]
        for u,v,w in edges:
            gr[u].append(v)
            gr[v].append(u)
        compo = defaultdict(int)
        compno = 1
        vis = [0]*n

        def dfs(node,gr,vis,compno,compo):
            compo[node] = compno
            vis[node] = 1
            for i in gr[node]:
                if vis[i] == 0:
                    dfs(i,gr,vis,compno,compo)

        for i in range(n):
            if vis[i] == 0:
                dfs(i,gr,vis,compno,compo)
                compno+=1
        score = [2**31 - 1]*n
        for u,v,w in edges:
            score[compo[u]] = score[compo[u]]&w
        ans = []
        for u,v in query:
            if u ==v :
                ans.append(0)
            elif compo[u] == compo[v]:
                ans.append(score[compo[u]])
            else:
                ans.append(-1)
        return ans
```

```cpp
#include <vector>
#include <unordered_map>
#include <algorithm>
#include <limits>

class Solution {
public:
    std::vector<int> minimumCost(int n, std::vector<std::vector<int>>& edges, std::vector<std::vector<int>>& query) {
        std::vector<std::vector<int>> gr(n);
        for (const auto& edge : edges) {
            int u = edge[0], v = edge[1];
            gr[u].push_back(v);
            gr[v].push_back(u);
        }

        std::unordered_map<int, int> compo;
        int compno = 1;
        std::vector<int> vis(n, 0);

        auto dfs = [&](int node, const std::vector<std::vector<int>>& gr, std::vector<int>& vis, int compno, std::unordered_map<int, int>& compo) {
            compo[node] = compno;
            vis[node] = 1;
            for (int i : gr[node]) {
                if (vis[i] == 0) {
                    dfs(i, gr, vis, compno, compo);
                }
            }
        };

        for (int i = 0; i < n; ++i) {
            if (vis[i] == 0) {
                dfs(i, gr, vis, compno, compo);
                compno++;
            }
        }

        std::vector<int> score(n, std::numeric_limits<int>::max());
        for (const auto& edge : edges) {
            int u = edge[0], v = edge[1], w = edge[2];
            score[compo[u]] &= w;
        }

        std::vector<int> ans;
        for (const auto& q : query) {
            int u = q[0], v = q[1];
            if (u == v) {
                ans.push_back(0);
            } else if (compo[u] == compo[v]) {
                ans.push_back(score[compo[u]]);
            } else {
                ans.push_back(-1);
            }
        }
        return ans;
    }
};
```

```java
import java.util.*;

public class Solution {
    public List<Integer> minimumCost(int n, int[][] edges, int[][] query) {
        List<List<Integer>> gr = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            gr.add(new ArrayList<>());
        }
        for (int[] edge : edges) {
            int u = edge[0];
            int v = edge[1];
            gr.get(u).add(v);
            gr.get(v).add(u);
        }

        Map<Integer, Integer> compo = new HashMap<>();
        int compno = 1;
        int[] vis = new int[n];

        // dfs function
        Stack<Integer> stack = new Stack<>();
        BiConsumer<Integer, Integer> dfs = (node, compnoArg) -> {
            compo.put(node, compnoArg);
            vis[node] = 1;
            stack.push(node);
            while (!stack.isEmpty()) {
                int curr = stack.pop();
                for (int i : gr.get(curr)) {
                    if (vis[i] == 0) {
                        stack.push(i);
                        vis[i] = 1;
                        compo.put(i, compnoArg);
                    }
                }
            }
        };

        for (int i = 0; i < n; i++) {
            if (vis[i] == 0) {
                dfs.accept(i, compno);
                compno++;
            }
        }

        int[] score = new int[n];
        Arrays.fill(score, Integer.MAX_VALUE);
        for (int[] edge : edges) {
            int u = edge[0];
            int v = edge[1];
            int w = edge[2];
            score[compo.get(u)] &= w;
        }

        List<Integer> ans = new ArrayList<>();
        for (int[] q : query) {
            int u = q[0];
            int v = q[1];
            if (u == v) {
                ans.add(0);
            } else if (compo.get(u).equals(compo.get(v))) {
                ans.add(score[compo.get(u)]);
            } else {
                ans.add(-1);
            }
        }
        return ans;
    }
}
```
